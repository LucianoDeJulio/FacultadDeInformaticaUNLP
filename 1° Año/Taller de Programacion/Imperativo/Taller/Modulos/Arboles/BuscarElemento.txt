procedure Buscar (a:arbol; var p:arbol;dato: integer);
begin
     if(a=nil) then
                p:=nil
     else
         if(dato=a^.dato)then
            p:=a
         else
             if(dato<a^.dato)then
                  Buscar(a^.HI,p,dato)
             else
                  Buscar(a^.HD,p,dato);
end;
