Procedure InsertarElemento ( var pri: lista; per: integer);
var ant, nue, act: lista;
begin
  new (nue);
  nue^.datos := per;
  act := pri;
  ant := pri;

  while (act<>NIL) and (act^.datos < per) do begin
      ant := act;
      act := act^.sig ;
  end;
  if (ant = act)  then pri := nue
                  else  ant^.sig  := nue;
  nue^.sig := act ;
end;
